10 {
11 ulong x = v & 0x1FFFFFul; // only use the lowest 21 bits, since we can only encode up to 2^21 - 1 in each axis
12 x = (x | (x << 32)) & 0x1F00000000FFFFul;

14 x = (x | (x << 8)) & 0x100F00F00F00F00Ful;

36 [MethodImpl(MethodImplOptions.AggressiveInlining)]
37 public static ulong Encode(int3 coord, byte depth)
38 {
39 // we can encode the 3d coordinate into a morton code by interleaving the bits of the x, y and z coordinates
40 ulong morton = ExpandBits((uint)coord.x) |
41 (ExpandBits((uint)coord.y) << 1) |
42 (ExpandBits((uint)coord.z) << 2);
43
44 // set the sentinel bit to the left of the last used bit to encode the depth
45 ulong sentinel = 1ul << (depth * 3);
46
47 return morton | sentinel;

53 [MethodImpl(MethodImplOptions.AggressiveInlining)]
54 public static int3 DecodeCoord(this ulong locationCode)
55 {
56 // remove the sentinel bit
57 int highestBitIndex = 63 - math.lzcnt(locationCode);
58 ulong mask = (1ul << highestBitIndex) - 1ul; // a mask that sets all bits below that sentinel bit to 1 and all bits above to 0
59 ulong rawMorton = locationCode & mask;
60
61 // now we can extract the x, y and z coordinates by compacting the bits. since we know that the x bits are at position 0, 3, 6... we can just shift the raw morton code to the right by 0, 1 and 2 bits and then compact the bits to get the original coordinates.
62 return new int3(
63 (int)CompactBits(rawMorton),
64 (int)CompactBits(rawMorton >> 1),
65 (int)CompactBits(rawMorton >> 2)
66 );
