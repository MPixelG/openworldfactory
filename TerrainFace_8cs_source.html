39 // Splits a triangle into many smaller triangles
40 // based on the given resolution.
41 private static (Vector3[], int[]) TriangleFragmentation(Vector3[] triangle, int res)
42 {
43 // Total vertex count for a triangular grid
44 Vector3[] vertices = new Vector3[(res + 1)*(res + 2) / 2];
45 // Each subdivision creates small triangles
46 int[] triangles = new int[res * res*3];
47 // Original triangle corners.
48 Vector3 a = triangle[2];
49 Vector3 b = triangle[1];
50 Vector3 c = triangle[0];
51 // Distance between subdivision steps in barycentric space.
52 float step = 1f / res;
53 int vertexIndex = 0;
54 int indexPos = 0;
55 // Generate all subdivided vertices row by row.
56 // Uses barycentric interpolation to stay inside the triangle.
57 for (int row = 0; row <= res; row++)
58 {
59 for (int col = 0; col <= res-row; col++)
60 {
61 float u = col*step;
62 float v = row*step;
63 // Interpolate between triangle corners.
64 vertices[vertexIndex++] = (1f-u-v)*a+u*b+v*c;
65 }
66 }
67
68 // Build triangle indices for the mesh.
69 for (int row = 0; row < res; row++)
70 {
71 for (int col = 0; col < res-row; col++)
72 {
73 // First triangle in the current grid cell.
74 triangles[indexPos++] = VertexIndex(row, col);
75 triangles[indexPos++] = VertexIndex(row, col+1);
76 triangles[indexPos++] = VertexIndex(row+1, col);
77 // Second triangle (if the cell is not on the edge).
78 if (col+1<res-row)
79 {
80 triangles[indexPos++] = VertexIndex(row, col+1);
81 triangles[indexPos++] = VertexIndex(row+1, col+1);
82 triangles[indexPos++] = VertexIndex(row+1, col);
83 }
84 }
85 }
86 return (vertices, triangles);
87
88 // Converts triangular grid coordinates (row, col)
89 // into a flat array index.
90 int VertexIndex(int row, int col)
91 {
92 return row * (res + 1) - row * (row - 1) / 2 + col;
93 }
94 }

104 {
105 for (int x = 0; x < _resolution; x++)

108 // Convert grid position into 0-1 range
109 Vector2 percent = new Vector2(x,y)/(_resolution-1);
110 // Position on cube face
111 Vector3 pointOnUnitCube = _localUp+(percent.x - 0.5f)*2*_axisA+(percent.y - 0.5f)*2*_axisB;
112 // Normalize to project onto sphere
113 Vector3 pointOnUnitSphere = pointOnUnitCube.normalized;
114 vertices[i] = pointOnUnitSphere;
115 // Generate quad triangles
116 if (x != _resolution - 1 && y != _resolution - 1)
117 {
118 triangles[triIndex] = i;
119 triangles[triIndex+1] = i+_resolution+1;
120 triangles[triIndex+2] = i+_resolution;
121 triangles[triIndex+3] = i;
122 triangles[triIndex+4] = i+1;
123 triangles[triIndex+5] = i+_resolution+1;
124 triIndex += 6;
125 }
126 }
127 }
