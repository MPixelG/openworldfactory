171 private static int GetCubeIndexAt(int3 pos, DensityFieldData grid, float isoLevel)
172 {
173 int cubeIndex = 0;
174
175 if (grid.DensityAt(pos.x, pos.y, pos.z) > isoLevel)
176 cubeIndex |=
177 1; // the |= operator sets every bit that is set in the right operand to 1 in the left operand also to 1.
178 // so if the left number (cubeIndex) is 0b11001000 and the right one (the mask, lets say 4) is 0b00000100 the result of that operation would be 0b11001100
179 if (grid.DensityAt(pos.x + 1, pos.y, pos.z) > isoLevel) cubeIndex |= 2;
180 if (grid.DensityAt(pos.x + 1, pos.y + 1, pos.z) > isoLevel) cubeIndex |= 4;
181 if (grid.DensityAt(pos.x, pos.y + 1, pos.z) > isoLevel) cubeIndex |= 8;
182 if (grid.DensityAt(pos.x, pos.y, pos.z + 1) > isoLevel) cubeIndex |= 16;
183 if (grid.DensityAt(pos.x + 1, pos.y, pos.z + 1) > isoLevel) cubeIndex |= 32;
184 if (grid.DensityAt(pos.x + 1, pos.y + 1, pos.z + 1) > isoLevel) cubeIndex |= 64;
185 if (grid.DensityAt(pos.x, pos.y + 1, pos.z + 1) > isoLevel) cubeIndex |= 128;
186
187 return cubeIndex;
188 }

199 // lerps (linear interpolates) between 2 given points based on their density values and the iso level
200 private static float3 VertexInterp(float isoLevel, int3 p1, int3 p2, float valP1, float valP2)
201 {
202 float3 p;
203
204 if (math.abs(isoLevel - valP1) < 0.00001)
205 return new float3(p1.x, p1.y, p1.z);
206 if (math.abs(isoLevel - valP2) < 0.00001)
207 return new float3(p2.x, p2.y, p2.z);
208 if (math.abs(valP1 - valP2) < 0.00001)
209 return new float3(p1.x, p1.y, p1.z);
210
211 float mu = (isoLevel - valP1) / (valP2 - valP1);
212 p.x = p1.x + mu * (p2.x - p1.x);
213 p.y = p1.y + mu * (p2.y - p1.y);
214 p.z = p1.z + mu * (p2.z - p1.z);
215
216 return p;
217 }
