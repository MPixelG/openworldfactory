52 private static int BuildNode(
53 ref Octree tree,
54 int3 min,
55 int3 max,
56 BurstSamplerSettings settings,
57 byte depth,
58 byte maxDepth
59 )
60 {
61 DensityFieldData sample = DensityFieldBuilder.BuildBurstDensityFieldDataInTree(
62 settings,
63 min,
64 max,
65 5 // we only need to sample the density at the corners and some samples in between to determine the state of the node. 5 is a good number for that.
66 );
67
68 float minDensity=float.PositiveInfinity;
69 float maxDensity=float.NegativeInfinity;
70 foreach (float densitySample in sample.Densities)
71 {
72 if(densitySample < minDensity) minDensity = densitySample;
73 if(densitySample > maxDensity) maxDensity = densitySample;
74 }
75
76 OctreeNodeState state = minDensity > BurstMeshGenerator.IsoLevel // if every value is above the isolevel the node is completely full
77 ? OctreeNodeState.Full
78 : maxDensity < BurstMeshGenerator.IsoLevel // if every value is below the isolevel the node is completely empty
79 ? OctreeNodeState.Empty
80 : OctreeNodeState.Mixed; // else the values are above and below the isolevel so its mixed
81
82 //todo if there are values below and above the isolevel in one quarter of a chunk we dont have to sample in that child chunk to see if its full empty since we know its mixed
83
84
85
86 OctreeNode node = new OctreeNode
87 {
88 Coord = min,
89 Depth = depth,
90 State = state,
91 FirstChildIndex = tree.Nodes.Length,
92 ChildMask = 0
93 };
94
95 int nodeIndex = tree.AddNode(node);
96
97 if (depth >= maxDepth || state == OctreeNodeState.Full || state == OctreeNodeState.Empty) // if we reached the max depth or the node is completely full or empty,
98 // we stop building and just add that node to the tree
99 {
100 return nodeIndex;
101 }
102
103 int3 size = max - min;
104 int3 half = size / 2;
105
106 for(int i = 0; i < 8; i++)
107 {
108 int3 offset = GetChildOffset(i);
109
110 int3 childMin =
111 min + offset * half;
112
113 int3 childMax =
114 childMin + half;
115
116 BuildNode(
117 ref tree,
118 childMin,
119 childMax,
120 settings,
121 (byte)(depth + 1),
122 maxDepth
123 );
124 }
125
126 node.ChildMask = 0xFF; // for now this is enough todo do better
127
128 tree.Nodes[nodeIndex] = node;
129
130 return nodeIndex;
131 }

156 public static float DensityAt(this Octree tree, DensityStorage densityStorage, float3 position)
157 {
158
159 float3 localPos = (position - tree.Min) / (tree.Max - tree.Min); // a position from 0|0|0 to 1|1|1 within the octree or below / above if the position is outside the octree
160
161 if (localPos.x < 0 || localPos.y < 0 || localPos.z < 0 || localPos.x > 1 || localPos.y > 1 ||
162 localPos.z > 1) return 0; // if the position is outside the bounds of the octree, return 0
163
164
165 OctreeNode currentNode = tree.Nodes[0]; // set the current node to the root node
166
167
168 float3 currentLocalPos = localPos; // the current local pos will get scaled every layer we go down to always be between 0 and 1 within that node
169 while (currentNode.ChildMask != 0)
170 {
171 // the child index offset is the index of the child we want to go to.
172 // we calculate it based on the current local position.
173 // if the x value is above 0.5, we want to go to the right half of the node, so we add 1 to the child index offset.
174 // if the y value is above 0.5, we want to go to the top half of the node, so we add 2 to the child index offset.
175 // and if the z value is above 0.5, we want to go to the back half of the node, so we add 4 to the child index offset.
176 // this way we can easily calculate the child index offset based on the current local position.
177 int childIndexOffset =
178 (currentLocalPos.x >= 0.5f ? 1 : 0) |
179 (currentLocalPos.y >= 0.5f ? 2 : 0) |
180 (currentLocalPos.z >= 0.5f ? 4 : 0);
181
182 int childIndex = currentNode.FirstChildIndex + childIndexOffset; // the final child index (the index in the complete node list) is calculated by just adding the offset to the first child index
183
184 currentNode = tree.Nodes[childIndex]; // set the current node to the child node (we traverse into that node)
185
186 currentLocalPos = math.frac(currentLocalPos * 2f); // now we just need to convert the local space of the parent into the local space of the child.
187 // we do this by multiplying the current local pos by 2.
188 // if the resulting position is above 1 we subtract 1.
189 // thats basically what the math.frac() function does.
190 }
191
192 OctreeNodeState currentNodeState = currentNode.State;
193 switch (currentNodeState)
194 {
195 case OctreeNodeState.Empty: // if the complete node only consists of density values below the isolevel we can just return 0 since it will be below the isolevel anyway
196 return 0;
197 case OctreeNodeState.Full: // the same goes for a full node (only containing values above the isolevel)
198 return 1;
199 case OctreeNodeState.Mixed:
200 {
201 NodeKey currentNodeKey = new NodeKey
202 {
203 Coord = currentNode.Coord,
204 Depth = currentNode.Depth
205 };
206
207 DensityFieldData densityField = densityStorage.GetDensityFieldDataOf(currentNodeKey);
208 float3 currentLocalPosUpscaled = currentLocalPos * densityField.Size; // currently the local pos is between 0 and 1. the positions in the density field range from 0 to the size of the density field,
209 // so we need to upscale the local pos to get the correct position in the density field.
210
211 return densityField.DensityAt(currentLocalPosUpscaled); // this DensityAt function (with a float3 as an input) does the interpolation for us
212 }
213
214 case OctreeNodeState.Unknown: // this shouldnt really happen (same goes for every other case) so we throw an exception.
215 default: throw new ArgumentOutOfRangeException();
216 }
217 }
218 }
