120 private void ProcessQueues()
121 {
122 int chunkCounter = 0; // counts the chunks we loaded or unloaded in this frame
123
124 while
125 (chunkCounter <
126 ChunksPerFrame) // we only want to load / unload a certain amount of chunks per frame to avoid performance spikes
127 {
128 if (_unloadQueue.Count <= 0) // break the loop if there are no more chunks to unload
129 break;
130
131 NodeKey coord = _unloadQueue.Dequeue(); // retrieve a chunk coord to unload from the queue
132
133 UnloadChunk(coord);
134 chunkCounter++;
135 }
136
137 while
138 (chunkCounter <
139 ChunksPerFrame) // we reuse the chunk counter without resetting it to ensure we only do a limited amount of operations (not loads and unloads) per frame
140 {
141 if (_activeGenerationTasks >=
142 MaxConcurrentGenerationTasks) // break if we cant start any more tasks this frame
143 break;
144
145 if (_loadQueue.Count <= 0) // or break if the queue is empty
146 break;
147
148 NodeKey coord = _loadQueue.Dequeue();
149
150 _ = GenerateChunkAsync(
151 coord); // we dont need the result of the task as its already automatically getting stored, but we need to add a reference to the task for it to start so we just use a tmp variable
152
153 chunkCounter++;
154 }

244 private void ProcessCompletedChunks()
245 {
246 while (_completedChunks.TryDequeue(out var result)) // every completed chunk gets processed
247 {
248 NodeKey coord = result.coord;
249 OctreeChunkData data = result.data;
250
251 if (!_chunkStates.TryGetValue(coord, out var state)) // if the state of the chunk doesnt exist, its broken as the state is already generated before starting the generation task. so to prevent further errors we just ignore that case
252 continue;
253
254 if (state != OctreeChunkLifecycleState.Generating) // the state should currently be in the "generating" phase
255 continue;
256
257 _chunks[coord] = data; // apply the data to the main chunks dictionary
258
259 _chunkStates[coord] = OctreeChunkLifecycleState.Loaded; // and update the state to loaded
260
261 NodeChange?.Invoke( // report the change to the listeners
262 new NodeChange(coord, Unity.NodeChange.NodeChangeType.Load)
263 );
264 }
