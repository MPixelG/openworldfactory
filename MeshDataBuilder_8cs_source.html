20 public void AddTriangle(float3 a, float3 b, float3 c)
21 {
22 // Skip triangles with invalid or degenerate data to avoid NaN normals.
23 if (!math.all(math.isfinite(a)) || !math.all(math.isfinite(b)) || !math.all(math.isfinite(c)))
24 {
25 return;
26 }
27
28 float3 ab = b - a;
29 float3 ac = c - a;
30 float3 normal = math.cross(ab, ac); // using the cross product of the 3 points we can get a rough estimation of the normals
31
32 if (math.lengthsq(normal) < 1e-12f) // if the length of the normal is invalid there probably was a division through zero somewhere
33 // so we just return (some really strange artifacts appear when there are vertices with NaN values in the mesh)
34 {
35 return;
36 }
37
38 int i0 = GetOrAddVertex(a);
39 int i1 = GetOrAddVertex(b); // get or add the vertices and get their indices. this way we dont always generate new indices and since they connect their normals connect too and the results are smoother.
40 int i2 = GetOrAddVertex(c);
41
42 _meshData.Indices.Add(i0);
43 _meshData.Indices.Add(i1); // add those indices to the mesh data
44 _meshData.Indices.Add(i2);
45
46 float angleA = AngleBetween(ab, ac);
47 float angleB = AngleBetween(c - b, a - b); // we calculate the angles between the edges of the triangle and use them to weight the normals. this way we get better results for triangles that are not equilateral and avoid artifacts in sharp corners.
48 float angleC = AngleBetween(a - c, b - c);
49
50 float3 faceNormal = math.normalize(normal); // normalize the normal so it has a length of 1. this way we can use it to calculate the vertex normals by adding it to the corresponding vertices and then normalizing them at the end.
51
52 _meshData.Normals[i0] += faceNormal * angleA;
53 _meshData.Normals[i1] += faceNormal * angleB; // weight the normals by the angles to get better results for non-equilateral triangles
54 _meshData.Normals[i2] += faceNormal * angleC;
55
56 }

63 private int GetOrAddVertex(float3 v)
64 {
65 int3 vertexKey = GetVertexKey(v); // calculate the vertex key of that vertex
66 if (_vertexMap.TryGetValue(vertexKey, out int index)) // if we already have a vertex with that key we return its index
67 return index;
68
69
70 index = _meshData.Vertices.Count; // otherwise we set the index to the current length of the vertices list
71
72 _meshData.Vertices.Add(v); // and add that vertex
73 _meshData.Normals.Add(float3.zero); // and normals
74
75 _vertexMap.Add(vertexKey, index); // finally add it to the vertex map
76

104 private static float AngleBetween(float3 a, float3 b)
105 {
106 float len = math.length(a) * math.length(b); // get the length
107
108 if (len < 1e-8f) // and check if its tiny, if it is we can skip the calculation since something that tiny will result in something even tinier and we can just return 0 to be more efficient
109 return 0f;
110
111 float d = math.clamp(math.dot(a, b) / len, -1f, 1f); // now just calculate the dot product and scale it down by the length.
112 // due to floating point inaccuracies it can happen that the value is slightly above 1 or below -1 so we just clamp it.
113
114 return math.acos(d); // finally we use acos to get the angle in radians
115 }
