97 private static float AngleBetween(float3 a, float3 b)
98 {
99 float len = math.length(a) * math.length(b); // get the length
100
101 if (len < 1e-8f) // and check if its tiny, if it is we can skip the calculation since something that tiny will result in something even tinier and we can just return 0 to be more efficient
102 return 0f;
103
104 float d = math.clamp(math.dot(a, b) / len, -1f, 1f); // now just calculate the dot product and scale it down by the length.
105 // due to floating point inaccuracies it can happen that the value is slightly above 1 or below -1 so we just clamp it.
106
107 return math.acos(d); // finally we use acos to get the angle in radians
108 }

111 private static float3 CalculateSmoothNormal(float3 pos, DensityField densityField)
112 {
113 // we sample slightly around the position to estimate the gradient
114 // this works like a derivative of the density field
115
116 float eps = 0.1f;
117
118
119 float dx =
120 densityField.DensityAt(pos + new float3(eps,0,0))
121 - densityField.DensityAt(pos - new float3(eps,0,0));
122
123 float dy =
124 densityField.DensityAt(pos + new float3(0,eps,0))
125 - densityField.DensityAt(pos - new float3(0,eps,0));
126
127 float dz =
128 densityField.DensityAt(pos + new float3(0,0,eps))
129 - densityField.DensityAt(pos - new float3(0,0,eps));
130
131 float3 gradient = new float3(dx, dy, dz);
132
133 // fallback in case gradient is zero
134 if (math.lengthsq(gradient) < 1e-12f)
135 {
136 return new float3(0, 1, 0);
137 }
138
139 // negative because density usually increases inward
140 return math.normalize(-gradient);
141 }
