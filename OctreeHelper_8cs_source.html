135 public static void Split(this Octree octree, int nodeIndex, BurstSamplerSettings settings, bool force=false)
136 {
137 if (nodeIndex == -1) throw new Exception("Node not found"); // if there is no node with that morton code, we throw an exception
138
139 OctreeNode node = octree.Nodes[nodeIndex]; // get the node
140
141 if (node.ChildMask != 0) return; // if that node already has children, we dont need to split it again
142
143 byte depth = node.MortonCode.GetDepth();
144 if (depth >= octree.MaxDepth && !force) return; // if we reached the max depth we cant split it anymore
145
146 BuildNode(ref octree, node.MortonCode, settings, (byte)(depth + 1)); // build the children of that node (this will automatically stop at one layer below the current depth and when a node is full or empty)
147
148 node.ChildMask = 0xFF;
149 octree.Nodes[nodeIndex] = node; // update the nodes values (currently only the child mask)
150 }

156

159 node.ChildMask = 0; // remove the children of that node by setting the child mask to 0
