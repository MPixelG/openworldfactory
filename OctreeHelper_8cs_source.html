59 private static void BuildNode(

88 ? OctreeNodeState.Full
89 : minDensity > BurstMeshGenerator.IsoLevel // if every value is above the isolevel the node is completely empty
90 ? OctreeNodeState.Empty
91 : OctreeNodeState.Mixed; // else the values are above and below the isolevel so its mixed

99 State = state,

134 public static void Split(this Octree octree, ulong mortonCode, BurstSamplerSettings settings, bool force=false)
135 {
136 bool containsValue = octree.IndexLookup.TryGetValue(mortonCode, out int nodeIndex); // get the position of the node with the given morton code in the node list by using the index lookup
137 bool outOfBounds = nodeIndex >= octree.Nodes.Length;
138 if(!containsValue || outOfBounds) return;
139
140
141 OctreeNode node = octree.Nodes[nodeIndex]; // get the node
142
143 if (node.ChildMask != 0) return; // if that node already has children, we dont need to split it again
144
145 byte depth = node.MortonCode.GetDepth();
146 if (depth >= octree.MaxDepth && !force) return; // if we reached the max depth we cant split it anymore (except if the user wants to)
147
148 BuildNode(ref octree, node.MortonCode, settings, (byte)(depth + 1)); // build the children of that node (this will automatically stop at one layer
149 // below the current depth and when a node is full or empty)
150
151 node.ChildMask = 0xFF;
152 octree.Nodes[nodeIndex] = node; // update the nodes values (currently only the child mask)
