89 private void RebuildOctreeBlocks()
90 {
91 if (_chunkManager == null || !_chunkManager.OctreeReady)
92 return;
93
94 _debugOctreeBlocks.Clear();
95 _instanceMatrices.Clear();
96
97 Octree octree = _chunkManager.Octree;
98 byte maxDepth = octree.MaxDepth;
99
100 foreach (OctreeNode node in octree.Nodes)
101 {
102 byte depth = node.MortonCode.GetDepth();
103
104 if (drawAllLayersUpToSelected)
105 {
106 if (depth > octreeGizmoDrawLayer) continue;
107 }
108 else
109 {
110 if (depth != octreeGizmoDrawLayer) continue;
111 }
112
113 if (node.State == OctreeNodeState.Full && !showFullStates ||
114 node.State == OctreeNodeState.Empty && !showEmptyStates ||
115 node.State == OctreeNodeState.Mixed && !showMixedStates ||
116 node.State == OctreeNodeState.Unknown && !showUnknownStates)
117 continue;
118
119 int nodeSize = 1 << (maxDepth - depth);
120
121 int3 localGridPos = node.MortonCode.DecodeCoord();
122
123 float3 worldMin = octree.Min + (localGridPos * nodeSize);
124
125 float3 center = worldMin + (new float3(nodeSize) * 0.5f);
126
127 var block = new DebugOctreeBlock
128 {
129 Center = center,
130 Size = nodeSize,
131 };
132
133 if (!_debugOctreeBlocks.TryGetValue(depth, out var layer))
134 {
135 layer = new List<DebugOctreeBlock>();
136 _debugOctreeBlocks.Add(depth, layer);
137 }
138
139 layer.Add(block);
140
141 _instanceMatrices.Add(
142 Matrix4x4.TRS(
143 block.Center,
144 Quaternion.identity,
145 Vector3.one * block.Size
146 )
147 );
148 }
149 }
