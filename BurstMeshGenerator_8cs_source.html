214 private static int GetCubeIndexAt(int3 pos, DensityFieldData grid, float isoLevel)
215 {
216 int cubeIndex = 0;
217
218 if (grid.DensityAt(pos.x, pos.y, pos.z) > isoLevel)
219 cubeIndex |=
220 1; // the |= operator sets every bit that is set in the right operand to 1 in the left operand also to 1.
221 // so if the left number (cubeIndex) is 0b11001000 and the right one (the mask, lets say 4) is 0b00000100 the result of that operation would be 0b11001100
222 if (grid.DensityAt(pos.x + 1, pos.y, pos.z) > isoLevel) cubeIndex |= 2;
223 if (grid.DensityAt(pos.x + 1, pos.y + 1, pos.z) > isoLevel) cubeIndex |= 4;
224 if (grid.DensityAt(pos.x, pos.y + 1, pos.z) > isoLevel) cubeIndex |= 8;
225 if (grid.DensityAt(pos.x, pos.y, pos.z + 1) > isoLevel) cubeIndex |= 16;
226 if (grid.DensityAt(pos.x + 1, pos.y, pos.z + 1) > isoLevel) cubeIndex |= 32;
227 if (grid.DensityAt(pos.x + 1, pos.y + 1, pos.z + 1) > isoLevel) cubeIndex |= 64;
228 if (grid.DensityAt(pos.x, pos.y + 1, pos.z + 1) > isoLevel) cubeIndex |= 128;
229
230 return cubeIndex;
231 }

234 // lerps (linear interpolates) between 2 given points based on their density values and the iso level
235 private static float3 VertexInterp(float isoLevel, int3 p1, int3 p2, float valP1, float valP2)
236 {
237 float3 p;
238
239 if (math.abs(isoLevel - valP1) < 0.00001)
240 return (new float3(p1.x, p1.y, p1.z));
241 if (math.abs(isoLevel - valP2) < 0.00001)
242 return (new float3(p2.x, p2.y, p2.z));
243 if (math.abs(valP1 - valP2) < 0.00001)
244 return (new float3(p1.x, p1.y, p1.z));
245
246 float mu = (isoLevel - valP1) / (valP2 - valP1);
247 p.x = p1.x + mu * (p2.x - p1.x);
248 p.y = p1.y + mu * (p2.y - p1.y);
249 p.z = p1.z + mu * (p2.z - p1.z);
250
251 return p;
252 }

254 private void AddTriangle(
255 float3 a, VertexKey ka,
256 float3 b, VertexKey kb,
257 float3 c, VertexKey kc)
258 {
259 if (!math.all(math.isfinite(a)) ||
260 !math.all(math.isfinite(b)) ||
261 !math.all(math.isfinite(c)))
262 return;
263
264 int i0 = GetOrAddVertex(ka, a);
265 int i1 = GetOrAddVertex(kb, b);
266 int i2 = GetOrAddVertex(kc, c);
267
268 Indices.Add(i0);
269 Indices.Add(i1);
270 Indices.Add(i2);
271
272 float3 normal = math.normalize(math.cross(b - a, c - a));
273
274 Normals[i0] += normal;
275 Normals[i1] += normal;
276 Normals[i2] += normal;
277 }

301 private void NormalizeNormals()
302 {
303 for (int i = 0; i < Normals.Length; i++)
304 {
305 float3 n = Normals[i];

307 {

309 continue;
310 }
311
312 Normals[i] = math.normalize(n);
