20 // we take that value from a huge precomputed table that contains the edge configuration for every possible corner configuration. So every bit is 1 if the isosurface cuts through that edge.

92 // returns the cube index of a given position in a given grid using the given iso level as a threshold. the cube index is an 8 bit integer where every bit represents whether that corner is inside (1) or outside (0) of the isosurface.
93 private static int GetCubeIndexAt(int3 pos, DensityField grid, float isoLevel)
94 {
95 int cubeIndex = 0;
96
97 if (grid.DensityAt(pos.x, pos.y, pos.z) > isoLevel) cubeIndex |= 1;// the |= operator sets every bit that is set in the right operand to 1 in the left operand also to 1.
98 // so if the left number (cubeIndex) is 0b11001000 and the right one (the mask, lets say 4) is 0b00000100 the result of that operation would be 0b11001100
99 if (grid.DensityAt(pos.x + 1, pos.y, pos.z) > isoLevel) cubeIndex |= 2;
100 if (grid.DensityAt(pos.x + 1, pos.y+1, pos.z) > isoLevel) cubeIndex |= 4;
101 if (grid.DensityAt(pos.x, pos.y+1, pos.z) > isoLevel) cubeIndex |= 8;
102 if (grid.DensityAt(pos.x, pos.y, pos.z+1) > isoLevel) cubeIndex |= 16;
103 if (grid.DensityAt(pos.x + 1, pos.y, pos.z+1) > isoLevel) cubeIndex |= 32;
104 if (grid.DensityAt(pos.x + 1, pos.y + 1, pos.z + 1) > isoLevel) cubeIndex |= 64;
105 if (grid.DensityAt(pos.x, pos.y + 1, pos.z + 1) > isoLevel) cubeIndex |= 128;
106
107 return cubeIndex;
108 }

111 // lerps (linear interpolates) between 2 given points based on their density values and the iso level
112 private static float3 VertexInterp(float isoLevel, int3 p1, int3 p2, float valP1, float valP2)
113 {
114 float3 p;
115
116 if (Math.Abs(isoLevel - valP1) < 0.00001)
117 return (new float3(p1.x, p1.y, p1.z));
118 if (Math.Abs(isoLevel - valP2) < 0.00001)
119 return (new float3(p2.x, p2.y, p2.z));
120 if (Math.Abs(valP1 - valP2) < 0.00001)
121 return (new float3(p1.x, p1.y, p1.z));
122
123 float mu = (isoLevel - valP1) / (valP2 - valP1);
124 p.x = p1.x + mu * (p2.x - p1.x);
125 p.y = p1.y + mu * (p2.y - p1.y);
126 p.z = p1.z + mu * (p2.z - p1.z);
127
128 return p;
129 }
