110 private void ProcessQueues()
111 {
112 int chunkCounter = 0; // counts the chunks we loaded or unloaded in this frame
113
114 while
115 (chunkCounter <
116 ChunksPerFrame) // we only want to load / unload a certain amount of chunks per frame to avoid performance spikes
117 {
118 if (_unloadQueue.Count <= 0) // break the loop if there are no more chunks to unload
119 break;
120
121 ChunkCoord coord = _unloadQueue.Dequeue(); // retrieve a chunk coord to unload from the queue
122
123 UnloadChunk(coord);
124 chunkCounter++;
125 }
126
127 while
128 (chunkCounter <
129 ChunksPerFrame) // we reuse the chunk counter without resetting it to ensure we only do a limited amount of operations (not loads and unloads) per frame
130 {
131 if (_activeGenerationTasks >=
132 MaxConcurrentGenerationTasks) // break if we cant start any more tasks this frame
133 break;
134
135 if (_loadQueue.Count <= 0) // or break if the queue is empty
136 break;
137
138 ChunkCoord coord = _loadQueue.Dequeue();
139
140 _ = GenerateChunkAsync(
141 coord); // we dont need the result of the task as its already automatically getting stored, but we need to add a reference to the task for it to start so we just use a tmp variable
142
143 chunkCounter++;
144 }

154 private void SyncChunks(float3 viewerPosition)
155 {
156 int3 viewerChunk =
157 (int3)math.floor(viewerPosition / ChunkSize); // convert world to chunk space
158
159 if (_hasLastViewerChunk && viewerChunk.Equals(_lastViewerChunk)) // if nothing changed since the last frames the results will be the same so we can exit
160 {
161 return;
162 }
163
164 _lastViewerChunk = viewerChunk;
165 _hasLastViewerChunk = true;
166
167 HashSet<ChunkCoord> visibleChunks =
168 _chunkStreamer.ComputeVisibleChunks(
169 viewerPosition,
170 _viewDistanceInChunks
171 ); // get the visible chunks calculated by the streamer
172
173 foreach (ChunkCoord coord in visibleChunks)
174 {
175 if (_chunkStates.TryGetValue(coord, out var state)) // try to get the current state of that chunk
176 {
177 if (state != ChunkLifecycleState.Unloaded) // if the chunk is anything else than unloaded (like loaded, generating etc.) we dont need to do anything
178 continue;
179 }
180
181 _chunkStates[coord] = ChunkLifecycleState.Queued; // set the state to queued
182
183 _loadQueue.Enqueue(coord); // append it to the load queue
184 }
185
186 HashSet<ChunkCoord> chunksToUnload = new(_chunks.Keys);
187
188 chunksToUnload.ExceptWith(visibleChunks); // every chunk thats currently loaded but doesnt need to be loaded gets unloaded
189
190 foreach (ChunkCoord coord in chunksToUnload)
191 {
192 if (!_chunkStates.TryGetValue(coord, out var state)) // if its state doesnt exist anymore it indicates that it probably got unloaded already
193 continue;
194
195 if (state != ChunkLifecycleState.Loaded) // if its currently unloaded, building, queued, etc. we ignore it
196 continue;
197
198 _chunkStates[coord] = ChunkLifecycleState.QueuedForUnload;
199
200 _unloadQueue.Enqueue(coord);
201 }
202 }

208 private async Task GenerateChunkAsync(ChunkCoord coord)
209 {
210 _activeGenerationTasks++; // keep track of the current amount of chunks that are being generated since they can generate across multiple frames
211
212 _chunkStates[coord] = ChunkLifecycleState.Generating; // set the current state to generating
213
214 try
215 {
216 ChunkData chunkData = await Task.Run(() => _chunkGenerator.GenerateChunkAt(coord)); // run the generation job
217
218 _completedChunks.Enqueue((coord, chunkData)); // and add its data afterward.
219 // this _completedChunks queue will get processed in the main thread in the update method to shift the
220 // generated data over to the main dictionaries and trigger the chunk change event
221 }
222 finally
223 {
224 _activeGenerationTasks--; // and finally (even if the task failed) remove it from the active tasks count
225 }
226 }

233 private void ProcessCompletedChunks()
234 {
235 while (_completedChunks.TryDequeue(out var result)) // every completed chunk gets processed
236 {
237 ChunkCoord coord = result.coord;
238 ChunkData data = result.data;
239
240 if (!_chunkStates.TryGetValue(coord, out var state)) // if the state of the chunk doesnt exist, its broken as the state is already generated before starting the generation task. so to prevent further errors we just ignore that case
241 continue;
242
243 if (state != ChunkLifecycleState.Generating) // the state should currently be in the "generating" phase
244 continue;
245
246 _chunks[coord] = data; // apply the data to the main chunks dictionary
247
248 _chunkStates[coord] = ChunkLifecycleState.Loaded; // and update the state to loaded
249
250 ChunkChange?.Invoke( // report the change to the listeners
251 new ChunkChange(coord, ChunkChangeType.Load)
252 );
253 }
254 }
